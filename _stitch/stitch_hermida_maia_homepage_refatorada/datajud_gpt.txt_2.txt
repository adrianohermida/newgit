✅ Tabela de Endpoints – Supabase (produção)
📌 Schema sugerido
api_catalog

1️⃣ Criação do schema
CREATE SCHEMA IF NOT EXISTS api_catalog;
ALTER SCHEMA api_catalog OWNER TO postgres;

2️⃣ Tabela principal: endpoints
CREATE TABLE IF NOT EXISTS api_catalog.endpoints (
    id uuid PRIMARY KEY DEFAULT gen_random_uuid(),

    -- Identificação
    nome text NOT NULL,
    descricao text,
    categoria text, -- ex: datajud, cnj, oab, receita, interno

    -- Endpoint
    metodo http_method NOT NULL,
    url text NOT NULL,
    versao text, -- ex: v1, v2
    ativo boolean DEFAULT true,

    -- Autenticação
    auth_tipo text NOT NULL, 
    -- ex: bearer, api_key, basic, oauth2, none

    auth_header text, 
    -- ex: Authorization, X-API-Key

    -- Parâmetros
    query_params jsonb,
    path_params jsonb,
    headers jsonb,
    body_schema jsonb,

    -- Resposta esperada
    response_schema jsonb,
    response_exemplo jsonb,

    -- Rate limit / regras
    rate_limit integer,
    timeout_ms integer DEFAULT 10000,

    -- Observações legais/técnicas
    base_legal text, -- ex: Resolução CNJ 331/2020
    observacoes text,

    -- Multi-tenant / controle
    tenant_id uuid,
    criado_por uuid,
    created_at timestamptz DEFAULT now(),
    updated_at timestamptz DEFAULT now()
);

3️⃣ Tipo ENUM para método HTTP (boa prática)
DO $$
BEGIN
    IF NOT EXISTS (SELECT 1 FROM pg_type WHERE typname = 'http_method') THEN
        CREATE TYPE http_method AS ENUM (
            'GET', 'POST', 'PUT', 'PATCH', 'DELETE'
        );
    END IF;
END$$;

4️⃣ Grants (Supabase)
GRANT USAGE ON SCHEMA api_catalog TO anon, authenticated;

GRANT SELECT
ON api_catalog.endpoints
TO anon;

GRANT SELECT, INSERT, UPDATE, DELETE
ON api_catalog.endpoints
TO authenticated;

5️⃣ RLS (obrigatório)
ALTER TABLE api_catalog.endpoints ENABLE ROW LEVEL SECURITY;

🔐 Política de leitura (pública ou autenticada)
CREATE POLICY "leitura_publica_ou_auth"
ON api_catalog.endpoints
FOR SELECT
USING (
    ativo = true
);

🔐 Escrita restrita (tenant / admin)
CREATE POLICY "escrita_autenticada"
ON api_catalog.endpoints
FOR INSERT
WITH CHECK (
    auth.role() = 'authenticated'
);

6️⃣ Índices recomendados
CREATE INDEX IF NOT EXISTS idx_endpoints_categoria
ON api_catalog.endpoints (categoria);

CREATE INDEX IF NOT EXISTS idx_endpoints_metodo
ON api_catalog.endpoints (metodo);

CREATE INDEX IF NOT EXISTS idx_endpoints_ativo
ON api_catalog.endpoints (ativo);

    p_codigo_tribunal text
)
RETURNS jsonb
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
    v_codigo text;
    v_result jsonb;
BEGIN
    -- Normalização do código
    v_codigo := upper(trim(p_codigo_tribunal));

    -- Normalizações comuns
    v_codigo := replace(v_codigo, '-', '');
    v_codigo := replace(v_codigo, ' ', '');

    -- Busca do endpoint
    SELECT jsonb_build_object(
        'codigo', nome,
        'descricao', descricao,
        'categoria', categoria,
        'metodo', metodo,
        'url', url,
        'auth_tipo', auth_tipo,
        'auth_header', auth_header,
        'base_legal', base_legal
    )
    INTO v_result
    FROM api_catalog.endpoints
    WHERE
        ativo = true
        AND replace(upper(nome), '-', '') = v_codigo
    LIMIT 1;

    -- Se não encontrado
    IF v_result IS NULL THEN
        RAISE EXCEPTION
            'Endpoint DataJud não encontrado para o tribunal: %',
            p_codigo_tribunal
            USING ERRCODE = 'P0001';
    END IF;

    RETURN v_result;
END;
$$;

1️⃣ Função SQL – Inferir Tribunal pelo Número do Processo (CNJ)
📌 Base técnica

O número CNJ segue o padrão:

NNNNNNN-DD.AAAA.J.TR.OOOO


J → ramo do Judiciário

TR → tribunal (região/UF)

Exemplo:

0001234-56.2023.8.26.0100
                ↑ ↑
                8 26  → TJSP

🔹 Função: inferir_codigo_tribunal_por_processo(npu)
CREATE OR REPLACE FUNCTION api_catalog.inferir_codigo_tribunal_por_processo(
    p_numero_processo text
)
RETURNS text
LANGUAGE plpgsql
IMMUTABLE
AS $$
DECLARE
    v_ramo text;
    v_tribunal text;
BEGIN
    -- remove caracteres não numéricos
    p_numero_processo := regexp_replace(p_numero_processo, '[^0-9]', '', 'g');

    -- valida tamanho mínimo
    IF length(p_numero_processo) < 14 THEN
        RAISE EXCEPTION 'Número de processo inválido: %', p_numero_processo;
    END IF;

    -- ramo (posição fixa)
    v_ramo := substring(p_numero_processo from 14 for 1);
    v_tribunal := substring(p_numero_processo from 15 for 2);

    -- Justiça Estadual
    IF v_ramo = '8' THEN
        RETURN 'TJ' || v_tribunal;
    END IF;

    -- Justiça Federal
    IF v_ramo = '4' THEN
        RETURN 'TRF' || v_tribunal;
    END IF;

    -- Justiça do Trabalho
    IF v_ramo = '5' THEN
        RETURN 'TRT' || v_tribunal;
    END IF;

    -- Justiça Eleitoral
    IF v_ramo = '6' THEN
        RETURN 'TRE-' || v_tribunal;
    END IF;

    -- Justiça Militar
    IF v_ramo = '7' THEN
        RETURN 'JM';
    END IF;

    -- Tribunais Superiores
    IF v_ramo = '9' THEN
        RETURN 'STJ';
    END IF;

    RAISE EXCEPTION 'Ramo do Judiciário não reconhecido: %', v_ramo;
END;
$$;

// supabase/functions/datajud-search/index.ts
import { serve } from "https://deno.land/std@0.224.0/http/server.ts"
import { createClient } from "https://esm.sh/@supabase/supabase-js@2"

serve(async (req) => {
  try {
    /* =============================
       1. Validação básica
    ============================== */
    if (req.method !== "POST") {
      return new Response(
        JSON.stringify({ error: "Método não permitido" }),
        { status: 405 }
      )
    }

    const body = await req.json()
    const {
      numero_processo,
      query,              // body DataJud
      search_after = null // paginação
    } = body

    if (!numero_processo || !query) {
      return new Response(
        JSON.stringify({ error: "numero_processo e query são obrigatórios" }),
        { status: 400 }
      )
    }

    /* =============================
       2. Supabase Client
    ============================== */
    const supabase = createClient(
      Deno.env.get("SUPABASE_URL")!,
      Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!
    )

    /* =============================
       3. Resolver endpoint DataJud
    ============================== */
    const { data: endpointData, error: endpointError } =
      await supabase.rpc(
        "get_datajud_endpoint_by_processo",
        { p_numero_processo: numero_processo }
      )

    if (endpointError || !endpointData?.endpoint?.url) {
      return new Response(
        JSON.stringify({
          error: "Falha ao resolver endpoint DataJud",
          details: endpointError
        }),
        { status: 500 }
      )
    }

    const endpointUrl: string = endpointData.endpoint.url

    /* =============================
       4. Montar payload DataJud
    ============================== */
    const payload: any = {
      ...query
    }

    if (search_after) {
      payload.search_after = search_after
    }

    /* =============================
       5. Chamada DataJud
    ============================== */
    const controller = new AbortController()
    const timeout = setTimeout(() => controller.abort(), 15000)

    const response = await fetch(endpointUrl, {
      method: "POST",
      headers: {
        "Authorization": `Bearer ${Deno.env.get("DATAJUD_API_KEY")}`,
        "Content-Type": "application/json"
      },
      body: JSON.stringify(payload),
      signal: controller.signal
    }).finally(() => clearTimeout(timeout))

    if (!response.ok) {
      const errorText = await response.text()
      return new Response(
        JSON.stringify({
          error: "Erro na API DataJud",
          status: response.status,
          body: errorText
        }),
        { status: 502 }
      )
    }

    const data = await response.json()

    /* =============================
       6. Resposta padronizada
    ============================== */
    return new Response(
      JSON.stringify({
        tribunal: endpointData.codigo_tribunal,
        endpoint: endpointUrl,
        total: data.hits?.total?.value ?? 0,
        hits: data.hits?.hits ?? [],
        search_after:
          data.hits?.hits?.length
            ? data.hits.hits[data.hits.hits.length - 1].sort
            : null
      }),
      { status: 200 }
    )

  } catch (err) {
    return new Response(
      JSON.stringify({
        error: "Erro inesperado",
        message: err instanceof Error ? err.message : String(err)
      }),
      { status: 500 }
    )
  }
})

4️⃣ Função SQL de Normalização (UPSERT completo)
CREATE OR REPLACE FUNCTION judiciario.normalizar_datajud(
  p_tenant_id uuid,
  p_datajud jsonb
)
RETURNS uuid
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  v_processo_id uuid;
BEGIN
  -- UPSERT processo
  INSERT INTO judiciario.processos (
    tenant_id,
    numero_cnj,
    classe,
    assunto,
    orgao_julgador,
    tribunal,
    ramo,
    grau,
    data_ajuizamento,
    valor_causa,
    segredo_justica,
    ultimo_movimento_em
  )
  VALUES (
    p_tenant_id,
    p_datajud->>'numeroProcesso',
    p_datajud#>>'{classeProcessual,nome}',
    p_datajud#>>'{assunto,nome}',
    p_datajud#>>'{orgaoJulgador,nome}',
    p_datajud->>'tribunal',
    p_datajud->>'ramo',
    (p_datajud->>'grau')::int,
    (p_datajud->>'dataAjuizamento')::date,
    (p_datajud->>'valorCausa')::numeric,
    (p_datajud->>'segredoJustica')::boolean,
    now()
  )
  ON CONFLICT (numero_cnj)
  DO UPDATE SET
    updated_at = now()
  RETURNING id INTO v_processo_id;

  -- Partes
  INSERT INTO judiciario.partes (processo_id, tenant_id, nome, tipo, polo, documento)
  SELECT
    v_processo_id,
    p_tenant_id,
    parte->>'nome',
    parte->>'tipoParte',
    parte->>'polo',
    parte->>'cpfCnpj'
  FROM jsonb_array_elements(p_datajud->'partes') parte;

  -- Movimentos
  INSERT INTO judiciario.movimentos (processo_id, tenant_id, codigo, descricao, data_movimento)
  SELECT
    v_processo_id,
    p_tenant_id,
    (mov->>'codigoMovimento')::int,
    mov->>'descricao',
    (mov->>'dataHora')::timestamptz
  FROM jsonb_array_elements(p_datajud->'movimentos') mov;

  RETURN v_processo_id;
END;
$$;

CREATE OR REPLACE VIEW judiciario.vw_processos_resumo AS

  p.tenant_id,
  date_trunc('month', m.data_movimento) AS mes,
  COUNT(*) AS total_movimentos

JOIN judiciario.processos p ON p.id = m.processo_id
GROUP BY p.tenant_id, mes
ORDER BY mes DESC;

🔹 2.5 – Tempo Médio entre Movimentos
CREATE OR REPLACE VIEW judiciario.vw_tempo_medio_movimentos AS
SELECT
  p.tenant_id,
  AVG(diff) AS media_dias
FROM (
  SELECT
    processo_id,
    EXTRACT(DAY FROM data_movimento -
      LAG(data_movimento) OVER (PARTITION BY processo_id ORDER BY data_movimento)
    ) AS diff
  FROM judiciario.movimentos
) t
JOIN judiciario.processos p ON p.id = t.processo_id
WHERE diff IS NOT NULL
GROUP BY p.tenant_id;

CREATE OR REPLACE FUNCTION lgpd.registrar_consentimento(
  p_tenant_id uuid,
  p_titular_id uuid,
  p_finalidade text,
  p_base_legal text,
  p_escopo text,
  p_origem text
)
RETURNS uuid
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE v_id uuid;
BEGIN
  INSERT INTO lgpd.consentimentos (

  )
  VALUES (
    p_tenant_id, p_titular_id, p_finalidade,

  )
  RETURNING id INTO v_id;

  RETURN v_id;
END;
$$;

🔹 3.2 – Verificar Consentimento (antes de qualquer tratamento)
CREATE OR REPLACE FUNCTION lgpd.tem_consentimento(
  p_tenant_id uuid,
  p_titular_id uuid,
  p_finalidade text
)
RETURNS boolean
LANGUAGE sql
AS $$
  SELECT EXISTS (
    SELECT 1
    FROM lgpd.consentimentos
    WHERE tenant_id = p_tenant_id
      AND titular_id = p_titular_id
      AND finalidade = p_finalidade
      AND consentido = true
      AND data_revogacao IS NULL
  );
$$;

1️⃣ VIEW CONSOLIDADA — “O que sabemos sobre o titular”
🔎 View: lgpd.vw_relatorio_titular
CREATE OR REPLACE VIEW lgpd.vw_relatorio_titular AS
SELECT
  c.titular_id,
  c.tenant_id,
  c.finalidade,
  c.base_legal,
  c.escopo,
  c.consentido,
  c.data_consentimento,
  c.data_revogacao,
  COUNT(l.id) AS total_operacoes,
  MIN(l.data_evento) AS primeiro_tratamento,
  MAX(l.data_evento) AS ultimo_tratamento
FROM lgpd.consentimentos c
LEFT JOIN lgpd.logs_tratamento l
  ON l.titular_id = c.titular_id
 AND l.finalidade = c.finalidade
GROUP BY
  c.titular_id, c.tenant_id, c.finalidade,
  c.base_legal, c.escopo,
  c.consentido, c.data_consentimento, c.data_revogacao;


✔️ Confirma existência de tratamento
✔️ Indica finalidade e base legal
✔️ Mostra período do tratamento

2️⃣ FUNÇÃO PRINCIPAL — Relatório Art. 18 (JSON oficial)
🔐 Função: lgpd.gerar_relatorio_titular_art18
CREATE OR REPLACE FUNCTION lgpd.gerar_relatorio_titular_art18(
  p_tenant_id uuid,
  p_titular_id uuid
)
RETURNS jsonb
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  v_relatorio jsonb;
BEGIN
  SELECT jsonb_build_object(
    'titular_id', p_titular_id,
    'confirmacao_tratamento', EXISTS (
      SELECT 1 FROM lgpd.logs_tratamento
      WHERE tenant_id = p_tenant_id
        AND titular_id = p_titular_id
    ),
    'dados_tratados', (
      SELECT jsonb_agg(
        jsonb_build_object(
          'finalidade', finalidade,
          'base_legal', base_legal,
          'escopo', escopo,
          'consentido', consentido,
          'data_consentimento', data_consentimento,
          'data_revogacao', data_revogacao,
          'total_operacoes', total_operacoes,
          'primeiro_tratamento', primeiro_tratamento,
          'ultimo_tratamento', ultimo_tratamento
        )
      )
      FROM lgpd.vw_relatorio_titular
      WHERE tenant_id = p_tenant_id
        AND titular_id = p_titular_id
    ),
    'direitos_titular', jsonb_build_array(
      'Confirmação do tratamento',
      'Acesso aos dados',
      'Correção',
      'Anonimização, bloqueio ou eliminação',
      'Portabilidade',
      'Informação sobre compartilhamento',
      'Revogação do consentimento'
    ),
    'data_emissao', now()
  )
  INTO v_relatorio;

  RETURN v_relatorio;
END;
$$;

CREATE OR REPLACE FUNCTION lgpd.portabilidade_titular(
  p_tenant_id uuid, p_titular_id uuid
)
RETURNS jsonb
LANGUAGE sql
AS $$
  SELECT jsonb_build_object(
    'consentimentos', (
      SELECT jsonb_agg(to_jsonb(c))
      FROM lgpd.consentimentos c
      WHERE tenant_id = p_tenant_id
        AND titular_id = p_titular_id

      SELECT jsonb_agg(to_jsonb(l))
      FROM lgpd.logs_tratamento l
      WHERE tenant_id = p_tenant_id
        AND titular_id = p_titular_id
    )
  );
$$;

🔹 3.3 Revogação de consentimento
CREATE OR REPLACE FUNCTION lgpd.revogar_consentimento(
  p_tenant_id uuid,
  p_titular_id uuid,
  p_finalidade text
)

LANGUAGE plpgsql
SECURITY DEFINER
AS $$

      data_revogacao = now()

    AND finalidade = p_finalidade;
END;
$$;
